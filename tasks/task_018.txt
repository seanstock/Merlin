```kotlin
@Composable
fun LockScreen(viewModel: LockScreenViewModel) {
    val state by viewModel.state.collectAsState()
    
    Box(modifier = Modifier.fillMaxSize()) {
        // Background
        Image(
            painter = painterResource(id = R.drawable.lock_background),
            contentDescription = null,
            modifier = Modifier.fillMaxSize(),
            contentScale = ContentScale.Crop
        )
        
        // Content
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(16.dp),
            verticalArrangement = Arrangement.Center,
            horizontalAlignment = Alignment.CenterHorizontally
        ) {
            // AI interaction or game content
            when (state.screenType) {
                ScreenType.AI_CHAT -> AIChatScreen(state.chatState)
                ScreenType.GAME -> GameContainer(state.gameState)
                ScreenType.TASK -> TaskScreen(state.taskState)
            }
            
            // Wallet display
            WalletDisplay(state.screenTimeBalance)
        }
    }
}
```
